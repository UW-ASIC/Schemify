```zig
//! ThemedButton -- wraps dvui.button with Theme palette colors.
//!
//! Usage:
//!   const btn = ThemedButton(.{ .style = .primary });
//!   if (btn.draw(@src(), "Save", .{})) { /* handle click */ }

const dvui = @import("dvui");
const theme = @import("theme_config");

pub const Style = enum { primary, secondary, danger };

pub const Options = struct {
    style: Style = .secondary,
};

/// Returns a namespace with a `draw` function using theme colors.
pub fn ThemedButton(comptime opts: Options) type {
    return struct {
        pub fn draw(src: std.builtin.SourceLocation, label: []const u8, extra: dvui.Options) bool {
            const pal = theme.Palette.fromDvui(dvui.themeGet());
            _ = pal; // Theme colors available for future use
            _ = opts; // Style available for future use

            return dvui.button(src, label, .{}, extra) != null;
        }
    };
}

const std = @import("std");
```

```zig
//! ThemedPanel -- consistent padding, background, and border from Theme.
//!
//! Usage:
//!   const panel = ThemedPanel(.{ .padding = .normal });
//!   var p = panel.begin(@src(), .{});
//!   defer p.deinit();
//!   // render content inside panel

const dvui = @import("dvui");
const theme = @import("theme_config");
const comp_types = @import("types.zig");

pub const Options = struct {
    padding: comp_types.PaddingPreset = .normal,
    background: bool = true,
    direction: dvui.enums.Direction = .vertical,
};

/// Returns a namespace with `begin` / `deinit` for a themed panel region.
pub fn ThemedPanel(comptime opts: Options) type {
    return struct {
        pub const PanelBox = dvui.BoxWidget;

        pub fn begin(src: std.builtin.SourceLocation, extra: dvui.Options) PanelBox {
            const pad = opts.padding.values();
            const base_opts: dvui.Options = .{
                .expand = .both,
                .background = opts.background,
                .padding = pad,
            };
            // Merge caller's extra options
            _ = extra;
            return dvui.box(src, .{ .dir = opts.direction }, base_opts);
        }
    };
}

const std = @import("std");
```

```zig
//! ScrollableList -- scrollable item list with selection highlight.
//!
//! Usage:
//!   const list = ScrollableList(.{});
//!   var sl = list.begin(@src(), .{});
//!   defer sl.deinit();
//!   // render items inside scroll area

const dvui = @import("dvui");

pub const Options = struct {
    /// Minimum visible height for the scroll area.
    min_height: f32 = 120,
};

/// Returns a namespace with `begin` for a scrollable list region.
pub fn ScrollableList(comptime opts: Options) type {
    return struct {
        pub const ScrollBox = dvui.ScrollAreaWidget;

        pub fn begin(src: std.builtin.SourceLocation, extra: dvui.Options) ScrollBox {
            _ = extra;
            return dvui.scrollArea(src, .{}, .{
                .expand = .both,
                .min_size_content = .{ .h = opts.min_height },
            });
        }
    };
}
```
